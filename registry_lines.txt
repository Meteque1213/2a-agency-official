        function renderTable(data) {
            const body = document.getElementById('registryBody');
            body.innerHTML = '';
            
            data.forEach(item => {
                // LOGIQUE DE COULEUR POUR LES NOTES (Version Claire)
                let scoreColor = 'text-green-600'; // Vert pro
                let statusText = 'VERIFIED';
                let statusClass = 'status-verified';

                if (item.score < 60) {
                    scoreColor = 'text-red-600'; // Rouge pro
                    statusText = 'CRITICAL';
                    statusClass = 'status-critical';
                } else if (item.score < 80) {
                    scoreColor = 'text-yellow-600'; // Jaune/Orange pro
                    statusText = 'WARNING';
                    statusClass = 'status-warning';
                }

                const row = `
                    <tr class="border-b border-gray-100 hover:bg-gray-50 transition-colors group">
                        <td class="py-6 font-semibold text-black">${item.name}</td>
                        <td class="py-6 text-gray-600 text-sm tracking-wide">${item.category}</td>
                        <td class="py-6 text-center">
                            <span class="${scoreColor} font-mono font-bold text-xl">${item.score}</span>
                            <span class="text-gray-300 text-xs">/100</span>
                        </td>
                        <td class="py-6">
                            <code class="text-xs text-gray-400 bg-gray-100 px-3 py-1.5 rounded-full border border-gray-200 font-mono">
                                ${item.tx}
                            </code>
                        </td>
                        <td class="py-6 text-right">
                            <span class="${statusClass} text-[10px] font-bold tracking-widest uppercase bg-opacity-10 px-3 py-1 rounded-full">
                                ${statusText}
                            </span>
                        </td>
                    </tr>
                `;
                body.innerHTML += row;
            });
        }

        function searchRegistry() {
            const input = document.getElementById('searchInput').value.toLowerCase();
            const filteredData = registryData.filter(item => 
                item.name.toLowerCase().includes(input) || 
                item.category.toLowerCase().includes(input)
            );
            renderTable(filteredData);
        }

        // Initialisation
        renderTable(registryData);
